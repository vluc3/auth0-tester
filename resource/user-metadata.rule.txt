function (user, context, callback) {
  var namespace = 'https://user_metadata_';
  user.user_metadata = user.user_metadata || {};
  
  context.accessToken[namespace + 'user_authorization']  = user.user_metadata; 
  context.idToken[namespace + 'user_authorization']  = user.user_metadata; 
  callback(null, user, context);
}
